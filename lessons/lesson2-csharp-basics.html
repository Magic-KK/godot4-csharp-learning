using Godot; public partial class Player : CharacterBody2D { // 属性 public int Health { get; set; } = 100; public float Speed { get; set; } = 200f; // 字段 private Vector2 _velocity; private bool _isDead = false; // 构造函数（Godot 中较少使用） public Player() { GD.Print("Player created!"); } }

继承

Godot C# 中，所有脚本都继承自 Godot 的基类：

// 2D 角色控制器 public partial class Player : CharacterBody2D { } // 3D 角色控制器 public partial class Player3D : CharacterBody3D { } // 普通节点 public partial class GameManager : Node { } // UI 控件 public partial class MainMenu : Control { }

接口

接口在 Godot 中用于定义行为规范：

// 定义接口 public interface IDamageable { void TakeDamage(int amount); void Heal(int amount); bool IsAlive { get; } } // 实现接口 public partial class Enemy : CharacterBody2D, IDamageable { private int _health = 100; public bool IsAlive => _health > 0; public void TakeDamage(int amount) { _health -= amount; GD.Print($"Enemy took {amount} damage, remaining: {_health}"); } public void Heal(int amount) { _health += amount; GD.Print($"Enemy healed {amount} HP"); } }
